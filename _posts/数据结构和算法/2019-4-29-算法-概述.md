    ```java
    int binarySearch(int R[], int low, int high, int k){
    	int mid = 0;
    	while(low <= high){
    		mid = (low + high)/2;
    		if (R[mid] == k) return mid;
    		else if (R[mid] > k) high = mid-1; // k位于mid位置的左边，high调整
    		else low = mid+1; // k位于mid位置的右边，low调整
    	}
    	return 0;
    }
    ```
